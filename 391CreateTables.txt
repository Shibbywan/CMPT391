CREATE TABLE Person (
SIN		INT,
FIRST NAME	VarChar(30),
LAST NAME	VarChar(30),
PRIMARY KEY(SIN));

CREATE TABLE CUSTOMER (
COUNTRY		VarChar(50),
CITY		VarChar(50),
ADDRESS		VarChar(50),
C_PHONE		VarChar(20),
POSTAL_CODE	VarChar(10),
C_ID		INT,
PRIMARY KEY(C_ID));

CREATE TABLE EMPLOYEE (
E_ID		INT,
POSITION	VarChar(20),
PRIMARY KEY(E_ID));

CREATE TABLE HOTEL (
NAME		VarChar(50),
RATING		VarChar(10),
H_ID		INT,
COUNTRY		VarChar(50),
CITY		VarChar(50),
ADDRESS		VarChar(50),
POSTAL_CODE	VarChar(10),
PHONE		VarChar(20),
NUM_OF_GUESTS	INT,
PRIMARY KEY(H_ID));

CREATE TABLE INVENTORY (
NAME		VarChar(50),
AMOUNT		INT,
ITEM_ID		INT,
H_ID		INT,
DESCRIPTION	VarChar(150),
PRIMARY KEY(ITEM_ID),
FOREIGN KEY(H_ID) REFERENCES HOTEL);

CREATE TABLE ROOMS (
R_ID		INT,
H_ID		INT,
PRICE		INT,
DESCRIPTION	VarChar(150),
NUM_OF_BEDS	INT,
SMOKING_ALLOWED	BIT,
ROOM_TYPE	VarChar(25),
FLOOR		INT,
PRIMARY KEY(R_ID),
FOREIGN KEY(H_ID) REFERENCES Hotel);

CREATE TABLE RESERVATION (
RES_ID		INT,
C_ID		INT,
R_ID		INT,
DATE		VarChar(15),
CHECK_OUT_TIME	VarChar(20),
CHECK_IN_TIME	VarChar(20),
PRIMARY KEY(RES_ID),
FOREIGN KEY(C_ID) REFERENCES CUSTOMER,
FOREIGN KEY(R_ID) REFERENCES ROOMS);

CREATE TABLE PAYMENT_METHOD (
M_ID		INT,
METHOD		VarChar(25),
PRIMARY KEY(M_ID));

CREATE TABLE INVOICE (
INV_ID		INT,
M_ID		INT,
C_ID		INT,
INV_DATE	VarChar(15),
WATER_COST	INT,
ELECTR_COST	INT,
SERVICES_COST	INT,
TOTAL_COST	INT,
PRIMARY KEY(INV_ID),
FOREIGN KEY(M_ID) REFERENCES PAYMENT_METHOD,
FOREIGN KEY(C_ID) REFERENCES CUSTOMER);

CREATE TABLE SERVICES (
NAME		VarChar(50),
S_ID		INT,
INV_ID		INT,
PRICE		INT,
DESCRIPTION	VarChar(100),
SERV_DATE	VarChar(15),
PRIMARY KEY(S_ID),
FOREIGN KEY(INV_ID) REFERENCES INVOICE);

CREATE TABLE CONTAINS (
INV_ID		INT,
M_ID		INT,
C_ID		INT,
SERV_DATE	VarChar(15),
INV_DATE	VarChar(15),
WATER_COST	INT,
ELECTR_COST	INT,
SERVICES_COST	INT,
TOTAL_COST	INT,
NAME		VarChar(50),
S_ID		INT,
PRICE		INT,
DESCRIPTION	VarChar(100),
PRIMARY KEY(S_ID, INV_ID),
FOREIGN KEY(M_ID) REFERENCES PAYMENT_METHOD,
FOREIGN KEY(C_ID) REFERENCES CUSTOMER
FOREIGN KEY(INV_ID) REFERENCES INVOICE);

CREATE TABLE PAYS_WITH (
M_ID		INT,
METHOD		VarChar(25),
INV_ID		INT,
C_ID		INT,
INV_DATE	VarChar(15),
WATER_COST	INT,
ELECTR_COST	INT,
SERVICES_COST	INT,
TOTAL_COST	INT,
PRIMARY KEY(M_ID, INV_ID),
FOREIGN KEY(M_ID) REFERENCES PAYMENT_METHOD,
FOREIGN KEY(C_ID) REFERENCES CUSTOMER,
FOREIGN KEY(INV_ID) REFERENCES INVOICE);

CREATE TABLE PAYS_FOR (
INV_ID		INT,
M_ID		INT,
C_ID		INT,
INV_DATE	VarChar(15),
WATER_COST	INT,
ELECTR_COST	INT,
SERVICES_COST	INT,
TOTAL_COST	INT,
COUNTRY		VarChar(50),
CITY		VarChar(50),
ADDRESS		VarChar(50),
PHONE		VarChar(20),
POSTAL_CODE	VarChar(10),
PRIMARY KEY(INV_ID,C_ID),
FOREIGN KEY(M_ID) REFERENCES PAYMENT_METHOD,
FOREIGN KEY(C_ID) REFERENCES CUSTOMER);

CREATE TABLE BOOKS (
COUNTRY		VarChar(50),
CITY		VarChar(50),
ADDRESS		VarChar(50),
PHONE		VarChar(20),
POSTAL_CODE	VarChar(10),
C_ID		INT,
RES_ID		INT,
R_ID		INT,
DATE		VarChar(15),
CHECK_OUT_TIME	VarChar(20),
CHECK_IN_TIME	VarChar(20),
PRIMARY KEY(RES_ID,C_ID),
FOREIGN KEY(C_ID) REFERENCES CUSTOMER,
FOREIGN KEY(R_ID) REFERENCES ROOMS);

CREATE TABLE STAYS_AT (
COUNTRY		VarChar(50),
C_CITY		VarChar(50),
ADDRESS		VarChar(50),
H_PHONE		VarChar(20),
POSTAL_CODE	VarChar(10),
C_ID		INT,
NAME		VarChar(50),
RATING		VarChar(10),
H_ID		INT,
COUNTRY		VarChar(50),
H_CITY		VarChar(50),
ADDRESS		VarChar(50),
POSTAL_CODE	VarChar(10),
C_PHONE		VarChar(20),
NUM_OF_GUESTS	INT,
PRIMARY KEY(C_ID, H_ID));

CREATE TABLE IN_STOCK (
NAME		VarChar(50),
RATING		VarChar(10),
H_ID		INT,
COUNTRY		VarChar(50),
CITY		VarChar(50),
ADDRESS		VarChar(50),
POSTAL_CODE	VarChar(10),
PHONE		VarChar(20),
NUM_OF_GUESTS	INT,
NAME		VarChar(50),
AMOUNT		INT,
ITEM_ID		INT,
H_ID		INT,
DESCRIPTION	VarChar(150),
PRIMARY KEY(H_ID,ITEM_ID),
FOREIGN KEY(H_ID));

CREATE TABLE HAS_ROOMS (
R_ID		INT,
H_ID		INT,
PRICE		INT,
DESCRIPTION	VarChar(150),
NUM_OF_BEDS	INT,
SMOKING_ALLOWED	BIT,
ROOM_TYPE	VarChar(25),
FLOOR		INT,
NAME		VarChar(50),
RATING		VarChar(10),
H_ID		INT,
COUNTRY		VarChar(50),
CITY		VarChar(50),
ADDRESS		VarChar(50),
POSTAL_CODE	VarChar(10),
PHONE		VarChar(20),
NUM_OF_GUESTS	INT,
PRIMARY KEY(H_ID,R_ID),
FOREIGN KEY(H_ID));
